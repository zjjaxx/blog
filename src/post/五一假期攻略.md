import { cli, Strategy } from '@jackwener/opencli/registry';
import { CliError } from '@jackwener/opencli/errors';

const DEFAULT_XCID = 'xc10eb6dada60700d1b7';
const DEFAULT_URL = `https://xuangu.eastmoney.com/Result?a=edit_way&id=${DEFAULT_XCID}`;
const DETAIL_URL = 'https://np-tjxg-b.eastmoney.com/api/smart-tag/stock/v3/getXcIdDetail';
const SEARCH_URL = 'https://np-tjxg-b.eastmoney.com/api/smart-tag/stock/v3/pw/search-code';

function makeRequestId() {
  return `${Date.now().toString(16)}${Math.random().toString(16).slice(2)}`
    .replace(/\./g, '')
    .slice(0, 32);
}

function getXcId(rawXcId, rawUrl) {
  if (rawXcId) return String(rawXcId).trim();
  if (!rawUrl) return DEFAULT_XCID;

  try {
    const url = new URL(String(rawUrl));
    const xcId = url.searchParams.get('id') || url.searchParams.get('xcId');
    if (xcId) return xcId;
  } catch {
    if (/^xc[a-z0-9]+$/i.test(String(rawUrl).trim())) return String(rawUrl).trim();
  }

  throw new CliError(
    'INVALID_ARGUMENT',
    'Unable to parse xcId from url. Pass --xcId directly or use a full Eastmoney Result URL.',
  );
}

async function fetchJson(url, init = {}) {
  const resp = await fetch(url, {
    ...init,
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json',
      'User-Agent': 'Mozilla/5.0',
      ...(init.headers || {}),
    },
  });

  if (!resp.ok) {
    throw new CliError('HTTP_ERROR', `Eastmoney request failed: HTTP ${resp.status}`, url);
  }

  return resp.json();
}

async function loadConfig(xcId) {
  const url = new URL(DETAIL_URL);
  url.searchParams.set('xcId', xcId);

  const data = await fetchJson(url);
  if (data?.code !== '100' || !data?.data) {
    throw new CliError('NO_DATA', `Unable to load screener config for ${xcId}`);
  }

  return data.data;
}

function buildSearchPayload(config, xcId, page, pageSize) {
  return {
    keyWord: config.contentNew || config.content || '',
    pageSize,
    pageNo: page,
    fingerprint: makeRequestId(),
    matchWord: '',
    shareToGuba: false,
    timestamp: Date.now(),
    requestId: makeRequestId(),
    removedConditionIdList: [],
    ownSelectAll: false,
    needCorrect: false,
    client: 'web',
    dxInfo: config.keywordInfoNew?.dxInfo || config.keywordInfo?.dxInfo || [],
    customData: config.customDataNew || config.customData || undefined,
    extraCondition: config.extraCondition || undefined,
    needShowStockNum: false,
    biz: 'web_ai_select_stocks',
    xcId,
  };
}

function pickConditionColumn(columns) {
  return (
    columns.find((col) => col?.userNeed === 1 && col?.key && col.key !== 'SERIAL') ||
    columns.find((col) => col?.sortWay && col?.key && col.key !== 'SERIAL') ||
    null
  );
}

function rowValue(row, key) {
  return key ? row?.[key] ?? '' : '';
}

cli({
  site: 'eastmoney',
  name: 'condition-picker',
  description: '东方财富条件选股结果页（默认最新股息率>5%）',
  domain: 'np-tjxg-b.eastmoney.com',
  strategy: Strategy.PUBLIC,
  browser: false,
  args: [
    { name: 'xcId', type: 'string', default: DEFAULT_XCID, help: '条件选股结果页 xcId' },
    { name: 'url', type: 'string', default: DEFAULT_URL, help: '东方财富条件选股结果页 URL；传入后会自动解析 xcId' },
    { name: 'page', type: 'int', default: 1, help: '页码，默认 1' },
    { name: 'pageSize', type: 'int', default: 50, help: '每页条数，默认 50，最大 100' },
  ],
  columns: [
    'rank',
    'code',
    'name',
    'market',
    'price',
    'changePercent',
    'conditionField',
    'conditionValue',
    'turnoverRate',
    'volume',
    'peDynamic',
    'pb',
    'totalMarketValue',
    'circulationMarketValue',
    'conditionText',
    'xcId',
  ],
  func: async (_page, args) => {
    const page = Math.max(1, Number(args.page) || 1);
    const pageSize = Math.max(1, Math.min(Number(args.pageSize) || 50, 100));
    const xcId = getXcId(args.xcId, args.url);

    const config = await loadConfig(xcId);
    const payload = buildSearchPayload(config, xcId, page, pageSize);
    const data = await fetchJson(SEARCH_URL, {
      method: 'POST',
      body: JSON.stringify(payload),
    });

    if (data?.code !== '100' || !data?.data?.result) {
      throw new CliError('NO_DATA', `Eastmoney returned no result for ${xcId}`, data?.msg);
    }

    const result = data.data.result;
    const columns = Array.isArray(result.columns) ? result.columns : [];
    const rows = Array.isArray(result.dataList) ? result.dataList : [];
    if (rows.length === 0) {
      throw new CliError('NO_DATA', `Eastmoney screener returned empty result for ${xcId}`);
    }

    const conditionColumn = pickConditionColumn(columns);
    const offset = (page - 1) * pageSize;

    return rows.map((row, index) => ({
      rank: offset + index + 1,
      code: row.SECURITY_CODE || '',
      name: row.SECURITY_SHORT_NAME || '',
      market: row.MARKET_SHORT_NAME || '',
      price: row.NEWEST_PRICE || '',
      changePercent: row.CHG || '',
      conditionField: conditionColumn?.title || '',
      conditionValue: rowValue(row, conditionColumn?.key),
      turnoverRate: row.TURNOVER_RATE || '',
      volume: row.VOLUME || row.TRADING_VOLUMES || '',
      peDynamic: row.PE_DYNAMIC || '',
      pb: row.PB || '',
      totalMarketValue: row['TOAL_MARKET_VALUE<140>'] || row.TOAL_MARKET_VALUE || '',
      circulationMarketValue: row['CIRCULATION_MARKET_VALUE<140>'] || row.CIRCULATION_MARKET_VALUE || '',
      conditionText: config.contentNew || config.content || '',
      xcId,
    }));
  },
});
